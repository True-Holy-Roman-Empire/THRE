every_country = {			
	limit = {
		religion = abrahamism
		NOT = {
			has_country_flag = catholic_religion_flag
			has_country_flag = sunni_religion_flag
			has_country_flag = orthodox_religion_flag
			has_country_flag = coptic_religion_flag
		}
	}
	save_event_target_as = loggingcountry
	log = "[loggingcountry.GetName] Country is Abrahamic but is currently not being set to convert to another religion upon the Great Schism"
}

	limit = {

		}

		}
	}

		}

	}
}
